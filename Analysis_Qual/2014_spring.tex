\section*{Part II: Real Analysis}

\subsection*{5}
Suppose $f \in L^{\infty}(\mathbb{R})$ and define the Laplace transform $F : (0,\infty) \rightarrow \mathbb{R}$ by 
\[ F(s) := \int_{0}^{\infty}f(t)e^{-s t}d\mu(t). \]
Prove that $F$ is absolutely continuous on $[a,b]$ for any $b > a > 0$.

{\bf Solution:}

\begin{Proof}
Let $\epsilon > 0$ and $b > a > 0$. Since $f \in L^{\infty}(\mathbb{R})$, there exists $M > 0$ such that $f \leq M$ a.e. Let $\delta := \epsilon
a^{2}/M$ and suppose $\{ (x_{i}, y_{i}) \}_{i=0}^{n} \subseteq [a,b]$ are non-overlapping subintervals such that $\sum_{i=0}^{n}(y_{i} - x_{i}) < \delta$.
Then,
\begin{align*}
\sum_{i=0}^{n}|F(x_{i}) - F(y_{i})| & = \sum_{i=0}^{\infty}\left| \int_{0}^{\infty}f(t)[e^{-x_{i}t} - e^{-y_{i}t}]d\mu(t)\right| \\
& \leq M \sum_{i=0}^{n} \left[  \int_{0}^{\infty}e^{-x_{i}t}d\mu(t) - \int_{0}^{\infty}e^{-y_{i}t}d\mu(t)\right] \\
& = M \sum_{i=0}^{n}\left[ \frac{1}{x_{i}} - \frac{1}{y_{i}} \right] \\
& = M \sum_{i=0}^{n}\left[ \frac{y_{i} - x_{i}}{x_{i}y_{i}} \right] \leq \frac{M}{a^{2}} \sum_{i=0}^{\infty}(y_{i} - x_{i}) \leq \frac{M\delta}{a^{2}} = \epsilon.
\end{align*}
\end{Proof}

\section*{Part II: Real Analysis}

\subsection*{1}
Let $f$ be an unbounded real valued function defined on the open interval $(0,1)$. Is it possible that $f$ is uniformly continuous on $(0,1)$? 

{\bf Solution:} No.

\begin{Proof}
By way of contradiction, suppose $f : (0,1) \rightarrow \mathbb{R}$ is uniformly continuous yet unbounded. Let $\epsilon > 0$. Then there exists $\delta > 0$ such
that $|f(x) - f(y)| < \epsilon$ whenever $x,y \in (0,1)$ such that $|x - y| < \delta$. Now let $0 < a < b < 1$ such that $a < \delta$ and $1 - b <
\delta$. Then there exists $M > 0$ such that $|f(x)| < M$ for all $x \in [a,b]$ and therefore $|f(y)| < M + \epsilon$ for all $y \in (0,1) - [a,b]$.
So $|f(x)| < M + \epsilon$ for all $x \in (0,1)$. This is a contradiction.
\end{Proof}


\subsection*{3}
Let $p,q,r$ be positive real numbers with $\frac{1}{p} + \frac{1}{q} + \frac{1}{r} = 1$, and let $f \in L^{p}, g \in L^{q}, h \in L^{r}$. Prove that
$fgh \in L^{1}$ and 
\[ \|fgh\|_{1} \leq \|f\|_{p}\cdot \|g\|_{q}\cdot \|h\|_{r}.\]

{\bf Solution:}

\begin{Proof}
Let $s := qr / (q + r)$.

\begin{claim}
$g h \in L^{s}$ and $\|gh\|_{s} \leq \|g\|_{q}\cdot \|h\|_{r}$.
\end{claim}
\begin{claimproof}
Let $q' := (q + r) / r$ and $r' := (q + r) / q$. Clearly $q'$ and $r'$ are conjugate and $|g|^{s} \in L^{q'}$ and $|h|^{s} \in L^{r'}$. Thus, by H\"{o}lder's Inequality,
\[ \|gh\|_{s}^{s} = \int |g|^{s}\cdot |h|^{s}\ d\mu \leq \left( \int |g|^{q}\ d\mu \right)^{\frac{r}{q + r}}\left( \int |h|^{r}\ d\mu
\right)^{\frac{q}{q + r}} = \|g\|_{q}^{s}\cdot \|h\|_{r}^{s}. \]
\end{claimproof}

Now note that $p$ and $s$ are conjugate. Thus, by claim 1 and H\"{o}lder's applied to $f$ and $gh$,
\[ \|fgh\|_{1} \leq \|f\|_{p}\cdot \|gh\|_{s} \leq \|f\|_{p}\cdot \|g\|_{q} \cdot \|h\|_{r}. \]
\end{Proof}
